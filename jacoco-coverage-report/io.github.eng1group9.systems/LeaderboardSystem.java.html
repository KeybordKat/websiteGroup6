package io.github.eng1group9.systems;

import java.io.File;
import java.io.FileWriter;
import java.io.IOException;
import java.nio.file.Files;
import java.util.List;
import java.util.ArrayList;

/**
 * handles everything related to the leaderboard and storing scores
 */
public class LeaderboardSystem {
    
    /**
     * handles writing to the leaderboard
     * creates file if needed then offloads the formatting to another method and overwrites the file with that data
     * @param name
     * @param score
     */
    public static void updateLeaderboard(String name, int score) {
        try {
            String userHome = System.getProperty("user.home");
            String dirName = userHome + File.separator + "maze-game-team-6";
            File dir = new File(dirName);
            if (!dir.exists()) {
                dir.mkdirs();
            }

            File leaderboardFile = new File(dir, "Leaderboard.txt");
            if (!leaderboardFile.exists()) {
                leaderboardFile.createNewFile();
            }

            List<String> leaderboard = formatLeaderboard(name, score, leaderboardFile);

            FileWriter writer = new FileWriter(leaderboardFile, false);
            for (int i = 0; i < Math.min(5, leaderboard.size()); i++) {
                writer.write(leaderboard.get(i) + "\n");
            }
            writer.close();
        }
        catch (IOException e) {
            e.printStackTrace();
        }
    }

    /**
     * processes the score data from the leaderboard and the game just played
     * compares new score to other members of the leaderboard and inserts new player if applicable
     * @param name player name
     * @param score player score
     * @param file the leaderboard file
     * @return the leaderboard list containing the 5 highest scoring players
     * @throws IOException
     */
    public static List<String> formatLeaderboard(String name, int score, File file) throws IOException {
        List<String> currentLeaders = Files.readAllLines(file.toPath());
        String newEntry = name + " - " + score;
        boolean inserted = false;

        for (int i = 0; i < currentLeaders.size(); i++) {
            String line = currentLeaders.get(i).trim();
            if (line.isEmpty()) continue;

            String[] parts = line.split(" - ", 2);
            if (parts.length < 2) continue;

            int existingScore = Integer.parseInt(parts[1].trim());
            if (score > existingScore) {
                currentLeaders.add(i, newEntry);
                inserted = true;
                break;
            }
        }

        if (!inserted) {
            currentLeaders.add(newEntry);
        }
        return currentLeaders;
    }

    public static List<String> getLeaderboardContents() {
        try {
            File file = getLeaderboardFile();
            if (!file.exists()) {
                return new ArrayList<String>();
            }
            return Files.readAllLines(file.toPath());
        }
        catch (IOException e){
            List<String> errorList = new ArrayList<String>();
            for (int i = 0; i < 5; i++) {
                errorList.add(e.toString());
            }
            return errorList;
        }
    }

    public static File getLeaderboardFile() {
        File leaderboardFile = new File(System.getProperty("user.home") + File.separator + "maze-game-team-6", "Leaderboard.txt");
        return leaderboardFile;
    }
}
