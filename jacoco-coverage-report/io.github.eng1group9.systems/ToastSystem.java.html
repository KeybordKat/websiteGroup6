package io.github.eng1group9.systems;

import java.io.File;
import java.io.FileWriter;
import java.io.IOException;
import java.nio.file.Files;
import java.util.LinkedList;
import java.util.List;

import com.badlogic.gdx.graphics.Color;

/**
 * System used to display temporary messages to the player.
 */
public class ToastSystem {

    /**
     * A toast is holds the data for one of the messages being displayed.
     * @param text - The text which should be displayed to the player.
     * @param createdDate - The time that this toast was created.
     * @param colour - The colour of the text used for this message.
     */
    static class Toast {
        private String text;
        private long createdDate;
        private Color colour = new Color(1, 1, 1, 1);
        private boolean writeToFile;
        private int bubbleWidth;
        private int padding;
        public boolean bubble;
        public int x;
        public int y;

        public Toast(String text, long createdDate) {
            this.text = text;
            this.createdDate = createdDate;
        }

        public Toast(String text, long createdDate, Color colour) {
            this.text = text;
            this.createdDate = createdDate;
            this.colour = colour;
        }

        public Toast(String text, long createdDate, Color colour, int bubbleWidth, int padding, int x, int y) {
            this.text = text;
            this.createdDate = createdDate;
            this.colour = colour;
            this.bubble = true;
            this.bubbleWidth = bubbleWidth;
            this.padding = padding;
            this.x = x;
            this.y = y;
    }

        public Toast(String text, long createdDate, boolean writeToFile) {
            this.text = text;
            this.createdDate = createdDate;
            this.writeToFile = writeToFile;
        }

        /**
         * @return The text assosiated with this toast.
         */
        public String getText() {
            return text;
        }

        /**
         * @return The colour of the text with this toast.
         */
        public Color getColour() {
            return colour;
        }

        public int getBubbleWidth() {
            return bubbleWidth;
        }

        public int getPadding() {
            return padding;
        }

        public int getX() {
            return x;
        }

        public int getY() {
            return y;
        }


        /**
         * Set the colour used when displaying the text.
         * @param colour
         */
        public void setColour(Color colour) {
            this.colour = colour;
        }

        /**
         * @return The time this toast was created.
         */
        public long getCreatedDate() {
            return createdDate;
        }
    }

    private List<Toast> toasts = new LinkedList<>(); // List of all active toasts.

    /**
     * Add a new toast to the systm, this will sent the user a new temporary message.
     * The colour will be White.
     * @param text - The text you want to display.
     */
    public void addToast(String text) {
        Toast toast = new Toast(text, System.currentTimeMillis());
        toasts.add(toast);
    }

    /**
     * Add a new toast to the systm, this will sent the user a new temporary message.
     * @param text - The text you want to display.
     * @param colour - The colour of the text.
     */
    public void addToast(String text, Color colour) {
        Toast toast = new Toast(text, System.currentTimeMillis(), colour);
        toasts.add(toast);
    }

    /**
     * add toast override for the achievement version, both prints to the screen and saves the relevant text to the achievment file
     * @param text the message to be displayed
     * @param writeToFile always true for given use case
     */
    public void addToast(String text, boolean writeToFile) {
        Toast toast = new Toast(text, System.currentTimeMillis(), writeToFile);
        toasts.add(toast);
        writeToFile(text.substring(text.indexOf(": ") + 2));
    }

    /**
     * add toast override for toasts that should appear in a speech bubble
     * @param text message to be displayed
     * @param textColor colour of the text
     * @param bubbleWidthPx size of the bubbles wrapping width
     * @param x bubbles x origin coord
     * @param y bubbles y origin coord
     */
    public void addToast(String text, Color textColor, int bubbleWidthPx, int x, int y) {
        Toast toast = new Toast(text, System.currentTimeMillis(), textColor, bubbleWidthPx, 12, x, y);
        toasts.add(toast);
    }
    /**
     * Look for toasts that have lived for more than 5s and delete them.
     */
    public void clearExpiredToasts() {
        Toast toastToRemove = null;

        for (Toast t : toasts) {
            if (System.currentTimeMillis() - t.getCreatedDate() > 5000) {
                toastToRemove = t;
            }
        }

        if (toastToRemove != null) {
            toasts.remove(toastToRemove);
        }
    }

    /**
     * @return A list of currently displayed toasts.
     */
    public List<Toast> getToasts() {
        return toasts;
    }

    /**
     * @return A list of colours in the same order as the toasts they are assosiated with.
     */
    public List<Color> getToastColourList() {
        List<Color> colourList = new LinkedList<>();
        for (Toast t : toasts) {
            colourList.add(t.getColour());
        }
        return colourList;
    }

    /**
     * writes a given toasts text into a file
     * in this use case it strips and formats text to just write ach names
     * @param text the message attached to the toast
    */
    public static void writeToFile(String text) {
        try {
            String userHome = System.getProperty("user.home");
            String dirName = userHome + File.separator + "maze-game-team-6";
            File dir = new File(dirName);
            if (!dir.exists()) {
                dir.mkdirs();
            }

            File achievementFile = new File(dir, "ach_list.txt");
            if (!achievementFile.exists()) {
                achievementFile.createNewFile();
            }

            List<String> achievements = Files.readAllLines(achievementFile.toPath());
            if (achievements.contains(text)) {
                return;
            }

            FileWriter writer = new FileWriter(achievementFile, true);
            writer.write(text + "\n");
            writer.close();
        }
        catch (IOException e) {
            e.printStackTrace();
        }
    }
}
